  <behavior>
Test group 1 — progress_percent generated column (service-role, no RLS):
  - Insert row with total_rows=100, processed_rows=50 → progress_percent=50
  - Insert row with total_rows=100, processed_rows=110 → progress_percent=100 (LEAST cap)
  - Insert row with total_rows=0, processed_rows=0 → progress_percent=0 (zero-division guard)
  - Attempt to INSERT providing progress_percent=99 explicitly → DB rejects (generated column)
  - Update processed_rows from 50→75 on existing row → progress_percent updates to 75

Test group 2 — contact_imports RLS isolation:
  - Org A user inserts a contact_imports row
  - Org B user queries contact_imports → receives 0 rows (invisible)
  - Org B user attempts to UPDATE org A's row → 0 rows affected
  - Org A user can read their own row → 1 row returned

Test group 3 — contact_import_errors RLS isolation (via import_id join):
  - Org A user inserts a contact_import_errors row linked to org A's import
  - Org B user queries contact_import_errors → receives 0 rows
  - Service-role can see both orgs' errors (bypasses RLS, used in cleanup)

Test group 4 — Realtime publication presence:
  - Query pg_publication_tables WHERE pubname='supabase_realtime' AND tablename='contact_imports'
    → returns exactly 1 row (confirms the ALTER PUBLICATION was applied)
  - contact_import_errors is NOT in the publication (it is fetched on-demand, not streamed)
    → query returns 0 rows for tablename='contact_import_errors'

Test group 5 — pg_cron job existence (IMP-18):
  - Query cron.job WHERE jobname='cleanup-stale-imports'
    → returns exactly 1 row
  - The schedule column equals '0 3 * * *'
  - The command contains 'contact_imports' and 'INTERVAL'

Test group 6 — ON DELETE CASCADE from contact_imports to contact_import_errors:
  - Service-role inserts a contact_imports row and two linked contact_import_errors rows
  - Service-role deletes the contact_imports row
  - Query contact_import_errors for the deleted import_id → 0 rows (CASCADE worked)

```typescript
/**
 * tests/import-schema.test.ts
 *
 * Schema-layer Vitest suite for Phase 73 (IMPORT-SCHEMA-WORKER).
 * Covers IMP-18 (cleanup cron) and IMP-19 (RLS isolation).
 *
 * Uses:
 * - Service-role client for setup/teardown and pg_publication_tables queries
 * - Org-scoped clients for RLS cross-org assertions
 *
 * Run: npx vitest run tests/import-schema.test.ts
 */
import { createClient } from '@supabase/supabase-js'
import { describe, it, expect, beforeAll, afterAll } from 'vitest'
import type { Database } from '../src/types/database'

// ----- clients ----------------------------------------------------------------

const SUPABASE_URL = process.env.NEXT_PUBLIC_SUPABASE_URL!
const SERVICE_KEY  = process.env.SUPABASE_SERVICE_ROLE_KEY!

const admin = createClient<Database>(SUPABASE_URL, SERVICE_KEY, {
  auth: { persistSession: false },
})

// ----- fixtures ---------------------------------------------------------------
// Use two real org UUIDs that exist on the remote DB. Look these up in the
// organizations table via service-role, or use UUIDs from earlier test suites
// (accounts-schema.test.ts / customfields-schema.test.ts established a pattern
// of querying the first two orgs from the table).

let orgAId: string
let orgBId: string
const createdImportIds: string[] = []
const createdErrorIds: string[]  = []

beforeAll(async () => {
  const { data: orgs } = await admin.from('organizations').select('id').limit(2)
  if (!orgs || orgs.length < 2) {
    throw new Error('Need at least 2 orgs in the remote DB for cross-org RLS tests')
  }
  orgAId = orgs[0].id
  orgBId = orgs[1].id
})

afterAll(async () => {
  // Clean up test rows (CASCADE deletes errors when imports are deleted)
  if (createdImportIds.length > 0) {
    await admin.from('contact_imports').delete().in('id', createdImportIds)
  }
})

// Helper: insert a minimal contact_imports row via service-role (bypasses RLS)
async function insertImport(orgId: string, overrides: Partial<Database['public']['Tables']['contact_imports']['Insert']> = {}) {
  const { data, error } = await admin
    .from('contact_imports')
    .insert({
      org_id: orgId,
      storage_path: `${orgId}/test-import/test.csv`,
      filename: 'test.csv',
      size_bytes: 1024,
      ...overrides,
    })
    .select('id, progress_percent, total_rows, processed_rows')
    .single()
  if (error) throw new Error(`insertImport failed: ${error.message}`)
  createdImportIds.push(data!.id)
  return data!
}
```
